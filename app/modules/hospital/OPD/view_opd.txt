select t0.*,t1.registerd_date,t1.patient_name,t1.gender,t1.blood_group, t1.dob,t1.age,t1.marital_status,t1.disability,t1.address,t1.location, t1.area, t1.city, t1.state, t1.country, t1.pincode, t1.guardian_details, t2.name from hospital_admission as t0 join hospital_patient as t1 on t1.id  = t0.patient_id join hospital_staff as t2 on t2.id = t0.staff_id
